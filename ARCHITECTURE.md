```sql
-- Scammer wallet tracking
CREATE TABLE scammer_wallets (
  wallet TEXT PRIMARY KEY,
  tokens_rugged INTEGER DEFAULT 0,
  total_stolen_usd DECIMAL DEFAULT 0,
  first_seen TIMESTAMP DEFAULT NOW(),
  last_seen TIMESTAMP DEFAULT NOW(),
  notes TEXT
);

-- Token outcome tracking for ML
CREATE TABLE token_outcomes (
  token TEXT PRIMARY KEY,
  launched_at TIMESTAMP,
  rugged_at TIMESTAMP,
  creator TEXT,
  bundle_detected BOOLEAN DEFAULT FALSE,
  initial_liquidity DECIMAL,
  peak_mcap DECIMAL
);

-- Agent activity log
CREATE TABLE agent_events (
  id SERIAL PRIMARY KEY,
  agent_type TEXT,
  event_type TEXT,
  token TEXT,
  data JSONB,
  created_at TIMESTAMP DEFAULT NOW()
);
```
